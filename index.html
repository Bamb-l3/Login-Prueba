// CONFIGURACIÓN DE SUPABASE const SUPABASE_URL = 'https://zustmsiqymroqgugjzqd.supabase.co'; // USA LA ANON KEY AQUÍ, NO LA SECRET KEY const SUPABASE_ANON_KEY = 'TU_ANON_PUBLIC_KEY'; const _supabase = supabase.createClient(SUPABASE_URL, SUPABASE_ANON_KEY); // LÓGICA DE LOGIN async function handleLogin(e) { e.preventDefault(); const email = document.getElementById('loginEmail').value; const password = document.getElementById('loginPassword').value; const errorMsg = document.getElementById('loginError'); // 1. Autenticación con Supabase Auth const { data, error } = await _supabase.auth.signInWithPassword({ email, password }); if (error) { showError(errorMsg, "Credenciales incorrectas: " + error.message); return; } // 2. Una vez logueado, buscamos su información en la tabla 'pacientes' // Asumimos que el ID de auth coincide con el UUID de tu tabla const { data: usuario, error: dbError } = await _supabase .from('pacientes') .select('*') .eq('id', data.user.id) .single(); if (dbError) { console.error("Error al obtener perfil:", dbError); } document.getElementById('login-view').classList.add('hidden'); // Redirección basada en el rol guardado en la tabla if (usuario && usuario.rol === 'clinic') { document.getElementById('clinic-view').classList.remove('hidden'); fillClinicForm(usuario); } else { document.getElementById('patient-view').classList.remove('hidden'); fillPatientForm(usuario); } } // Función para llenar el formulario del Paciente function fillPatientForm(data) { if (!data) return; document.getElementById('patName').value = data.nombre || ''; document.getElementById('patPhone').value = data.telefono || ''; document.getElementById('patDate').value = data.fecha_nacimiento || ''; document.getElementById('patBlood').value = data.tipo_sangre || 'A+'; document.getElementById('patNotes').value = data.notas || ''; } // Función para guardar cambios (UPSERT) async function saveProfile(e, type) { e.preventDefault(); const { data: { user } } = await _supabase.auth.getUser(); let updates = { id: user.id, nombre: type === 'patient' ? document.getElementById('patName').value : document.getElementById('cliName').value, telefono: type === 'patient' ? document.getElementById('patPhone').value : document.getElementById('cliPhone').value, // Agrega aquí los demás campos según tus columnas en Supabase }; const { error } = await _supabase.from('pacientes').upsert(updates); if (!error) { const toast = document.getElementById('toast'); toast.classList.remove('hidden'); setTimeout(() => toast.classList.add('hidden'), 3000); } else { alert("Error al guardar: " + error.message); } }
